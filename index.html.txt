<script>
const chat = document.getElementById("chat");
const cardsDiv = document.getElementById("cards");

let cards = [];
let index = 0;

/* Chat */
function sendMessage() {
  let input = document.getElementById("input");
  let text = input.value;
  if (!text) return;

  chat.innerHTML += `<div class="message user">${text}</div>`;

  let response = localAI(text);

  setTimeout(() => {
    chat.innerHTML += `<div class="message bot">${response}</div>`;
    chat.scrollTop = chat.scrollHeight;
  }, 300);

  input.value = "";
}

/* Local AI */
function localAI(text) {
  text = text.toLowerCase();

  if (text.includes("pete"))
    return "PETE = Point, Evidence, Technique, Explain.";

  if (text.includes("paper 1"))
    return "Paper 1 = Creative reading and writing.";

  if (text.includes("fraction"))
    return "Fractions represent parts of a whole.";

  return "Basic response. Add API for real intelligence.";
}

/* Subjects */
function loadEnglish() {
  chat.innerHTML = "<div class='message bot'>English ready: Paper 1, Paper 2, PETE, Writing.</div>";
}

function loadMaths() {
  chat.innerHTML = "<div class='message bot'>Maths ready: Fractions, Algebra, Ratio.</div>";
}

/* Flashcards */
function showCards() {
  cardsDiv.style.display = "block";
  chat.style.display = "none";

  cards = [
    {q:"PETE?", a:"Point, Evidence, Technique, Explain"},
    {q:"Paper 1?", a:"Creative reading & writing"},
    {q:"50% of 200?", a:"100"},
    {q:"3/4 decimal?", a:"0.75"}
  ];

  index = 0;
  renderCard();
}

function renderCard() {
  let c = cards[index];

  cardsDiv.innerHTML = `
    <div class="card" onclick="flipCard()">
      <div id="front">${c.q}</div>
      <div id="back" style="display:none">${c.a}</div>
    </div>

    <div class="controls">
      <button onclick="know()">I Know</button>
      <button onclick="dontKnow()">Don't Know</button>
      <button onclick="nextCard()">Next</button>
    </div>
  `;
}

function flipCard() {
  let f = document.getElementById("front");
  let b = document.getElementById("back");

  if (b.style.display === "none") {
    b.style.display = "block";
    f.style.display = "none";
  } else {
    b.style.display = "none";
    f.style.display = "block";
  }
}

function nextCard() {
  index = (index + 1) % cards.length;
  renderCard();
}

/* Progress tracking */
function know() {
  saveProgress(true);
  nextCard();
}

function dontKnow() {
  saveProgress(false);
  nextCard();
}

function saveProgress(result) {
  let stats = JSON.parse(localStorage.getItem("progress")) || {good:0, bad:0};

  if (result) stats.good++;
  else stats.bad++;

  localStorage.setItem("progress", JSON.stringify(stats));
}

/* Tools */
function summarise() {
  let text = document.getElementById("input").value;
  chat.innerHTML += `<div class="message bot">Summary: ${text.slice(0,100)}...</div>`;
}

function modelAnswer() {
  chat.innerHTML += `<div class="message bot">
  Structured Answer:<br>
  Point → Answer<br>
  Evidence → Example<br>
  Explain → Detail
  </div>`;
}
